DROP TABLE IF EXISTS job;
CREATE TABLE IF NOT EXISTS job(
dateCreation VARCHAR(1000)  ,
qualificationCode VARCHAR(1000)  ,
dateActualisation DATETIME  ,
dureeTravailLibelle VARCHAR(1000)  ,
intitule VARCHAR(1000)  ,
typeContratLibelle VARCHAR(1000)  ,
secteurActiviteLibelle VARCHAR(1000)  ,
description TEXT  ,
origineOffre VARCHAR(1000)  ,
lieuTravail_id VARCHAR(100)  ,
experienceLibelle VARCHAR(1000)  ,
romeLibelle VARCHAR(1000)  ,
nombrePostes VARCHAR(1000)  ,
accessibleTH VARCHAR(1000)  ,
codeNAF VARCHAR(1000)  ,
contact_id VARCHAR(100) NOT NULL ,
qualificationLibelle VARCHAR(1000)  ,
dureeTravailLibelleConverti VARCHAR(1000)  ,
agence VARCHAR(1000)  ,
alternance VARCHAR(1000)  ,
typeContrat VARCHAR(1000)  ,
secteurActivite VARCHAR(1000)  ,
competence_id VARCHAR(100) NOT NULL ,
appellationlibelle VARCHAR(1000)  ,
permis VARCHAR(1000)  ,
natureContrat VARCHAR(1000)  ,
entreprise_id VARCHAR(100) NOT NULL ,
experienceExige VARCHAR(1000)  ,
id VARCHAR(100) NOT NULL ,
offresManqueCandidats BOOLEAN  ,
salaire_id VARCHAR(100) NOT NULL ,
romeCode VARCHAR(1000)  ,
PRIMARY KEY ( id )
CONSTRAINT lieuTravail_id REFRENCES lieuTravail(lieuTravail_id ),
CONSTRAINT contact_id REFRENCES contact(contact_id ),
CONSTRAINT competence_id REFRENCES competence(competence_id ),
CONSTRAINT entreprise_id REFRENCES entreprise(entreprise_id ),
CONSTRAINT salaire_id REFRENCES salaire(salaire_id ));
